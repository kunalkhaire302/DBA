Query 6: Average transaction amount per branch JOIN + AVG
SELECT b.branch_name, ROUND(AVG(t.amount), 2) AS avg_txn
FROM branches b
JOIN accounts a ON b.branch_id = a.branch_id
JOIN transactions t ON a.account_id = t.account_id
GROUP BY b.branch_name;
