Important SQL Queries Showcase

Query 1: List all customers with their account balance JOIN 
SELECT c.name, c.email, a.account_type, a.balance 
FROM customer c
INNER JOIN account a ON c.customer_id = a.customer_id;
Query 2: Total deposits & withdrawals per account GROUP BY + SUM
SELECT account_id, type, SUM(amount) AS total_volume
FROM transactions
GROUP BY account_id, type;
Query 3: Find accounts with high balance WHERE + ORDER BY 
SELECT account_id, customer_id, balance
FROM account
WHERE balance > 1000
ORDER BY balance DESC;
Query 4: Active accounts with multiple transactions HAVING 
SELECT account_id, COUNT(*) AS txn_count
FROM transactions
GROUP BY account_id
HAVING COUNT(*) > 1;
Query 5: Top 5 highest balance accounts LIMIT
SELECT account_id, balance
FROM account
ORDER BY balance DESC
LIMIT 5;
Query 6: Average transaction amount per account AVG
SELECT account_id, ROUND(AVG(amount), 2) AS avg_txn
FROM transactions
GROUP BY account_id;
Query 7: Find dormant accounts (no transactions) LEFT JOIN + NULL
SELECT a.account_id, a.balance
FROM account a
LEFT JOIN transactions t ON a.account_id = t.account_id
WHERE t.transaction_id IS NULL;
Query 8: Monthly transaction summary DATE_FORMAT
SELECT DATE_FORMAT(date, '%b-%Y') AS month, SUM(amount) AS total
FROM transactions
GROUP BY month;
Query 9: Customers with multiple account types SUBQUERY
SELECT name FROM customer WHERE customer_id IN (
    SELECT customer_id FROM account
    GROUP BY customer_id
    HAVING COUNT(DISTINCT account_type) > 1
);
Query 10: Find duplicate phones GROUP BY + HAVING
SELECT phone, COUNT(*) AS count
FROM customer
GROUP BY phone
HAVING COUNT(*) > 1;
Query 11: System Audit Log Overview AUDIT
SELECT * FROM audit_log LIMIT 10;
Query 12: Show optimization plan EXPLAIN
EXPLAIN SELECT * FROM account WHERE customer_id = 1;
Query 13: List table performance indexes INDEXES
SHOW INDEX FROM transactions;
Query 14: Show user privileges Matrix PRIVILEGES
SELECT * FROM user_privileges LIMIT 10;
Query 15: Select recent silent mutation logs AUDIT RECENT 
SELECT * FROM audit_log 
ORDER BY performed_at DESC 
LIMIT 3;
